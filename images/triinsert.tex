\begin{algorithm}[H]
\SetAlgoRefName{4}
\caption{Tri par insertion}%
\Donnees{T un tableau de $n$ entiers}
\Res{Le tableau T est trié par ordre croissant}
\Pour{i $\leftarrow$ 2 à n} {
    clé $\leftarrow T[i]$\;
    $j \leftarrow  i-1$\;
    \Tq{$j>0$ et $T[j]>$clé}{
    $T[j+1] \leftarrow T[j]$\;
    $j \leftarrow j-1$\;
    }
    $T[j+1] \leftarrow$ clé\;
}
\Return T\;
\end{algorithm}%
