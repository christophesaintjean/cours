\begin{algorithm}[H]
\SetAlgoRefName{1}
\caption{Calcul de l'élément maximal}%
\Donnees{T un tableau de n entiers}
\Res{$i_{\max}$ et $\max$ tel que $T[i_{\max}]=\max$ avec max comme la plus grande valeur de T}
$i_{\max} \leftarrow 1$\;
$\max \leftarrow T[1]$\;
\Pour{i $\leftarrow$ 2 à n}
{\Si{$T[i]>\max$}{$\max \leftarrow T[i]$\;$i_{\max} \leftarrow i$\;}}
\end{algorithm}%
