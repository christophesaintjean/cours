\begin{algorithm}[H]
\caption{Calcul de l'élément maximal d'un tableau trié}%
\SetAlgoRefName{321} 
\Donnees{T un tableau trié dans l'ordre croissant de $n$ entiers}
\Res{$i_{\max}$ et $\max$ tel que $T[i_{\max}]=\max$ avec max comme la plus grande valeur de T}
$i_{\max} \leftarrow n$\;
$\max \leftarrow T[n]$\;
\end{algorithm}%
